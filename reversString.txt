package edu.hello;

public class ReversString
{
	public static void main(String[] args)	
	{
		String s = "hello hi man in the a given area";
		String rev= "";
		int count = 0;
		for (int i = s.length()-1; i >=0; i--) 
		{
			rev = rev+s.charAt(i);
			if (s.charAt(i)!='a') {
				continue;
			} else {
				count++;
			}
		}
		System.out.println(rev);
		System.out.println("count="+count);
	}
}
